# list-clis Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create a `list-clis` bash script at `~/.local/bin/list-clis` that scans `$PATH` and prints recognized CLIs organized by category.

**Architecture:** Single bash script with a `categorize()` function (case/glob patterns → category string) and a `main()` function (PATH scan → dedup → buffer → ordered print). No external dependencies beyond standard POSIX utilities.

**Tech Stack:** bash, standard POSIX file tests, `sort`, `sed`

```bash
#!/usr/bin/env bash
set -euo pipefail

categorize() {
  local name="$1"
  case "$name" in
    # AI & Editors
    claude|cursor|code|console-ninja|aider|copilot|cody|continue)
      echo "AI & Editors" ;;
    # Version Control
    git|git-*|gh|hub|hub-tool|scalar|jj|fossil|svn|hg)
      echo "Version Control" ;;
    # JavaScript / Node
    node|npm|npx*|nvm|pnpm|yarn|bun|deno|corepack|\
    tsc|tsserver|ts-node*|ts-script|nodemon|ncu|npm-check-updates|\
    eslint|prettier|jest|vitest|mocha|cypress|\
    webpack|rollup|esbuild|parcel|vite|turbo|nx|\
    create-vite|create-react-app|create-next-app|serve|json-server|express-*)
      echo "JavaScript / Node" ;;
    # Python
    python|python3*|pip*|uv|uvx|pipenv|poetry|pyenv|conda|mamba|\
    black|ruff|flake8|mypy|pylint|isort|pytest|sphinx-*)
      echo "Python" ;;
    # Go
    go|gofmt|gopls|golangci-lint|air|goreleaser)
      echo "Go" ;;
    # Rust
    rustc|rustup|rustfmt|rustdoc|cargo*|clippy-*|rust-*)
      echo "Rust" ;;
    # Ruby
    ruby|gem|bundle|bundler|rails|rake|rbenv|rvm|irb|pry)
      echo "Ruby" ;;
    # JVM
    java|javac|mvn|gradle|kotlin|kotlinc|scala|sbt|groovy)
      echo "JVM" ;;
    # Database
    psql*|pg_*|postgres*|pscale|createdb*|dropdb*|pgbench*|vacuumdb*|reindexdb*|\
    mysql|mysqldump|mariadb|sqlite3|redis-cli|mongosh|mongo)
      echo "Database" ;;
    # Containers & Orchestration
    docker*|kubectl*|helm|k9s|kind|minikube|k3s|podman|buildah|skopeo|containerd)
      echo "Containers & Orchestration" ;;
    # Cloud & Infrastructure
    aws|gcloud|az|doctl|fly|flyctl|railway|heroku|\
    terraform|tofu|pulumi|ansible*|vagrant|packer|\
    vercel|netlify|wrangler|supabase)
      echo "Cloud & Infrastructure" ;;
    # Build Tools
    cmake|cpack|ctest|ninja|make|automake*|autoconf*|aclocal*|protoc*|bazel|meson|ant)
      echo "Build Tools" ;;
    # Package Managers
    brew|nix|snap|flatpak)
      echo "Package Managers" ;;
    # Utilities
    jq|yq|fzf|tmux|htop|btop|bat|eza|fd|rg|ngrok|openssl|op|stripe|z3|pa11y|swig*)
      echo "Utilities" ;;
    *)
      echo "" ;;
  esac
}

main() {
  declare -A buffers
  declare -A seen

  local -a category_order=(
    "AI & Editors"
    "Version Control"
    "JavaScript / Node"
    "Python"
    "Go"
    "Rust"
    "Ruby"
    "JVM"
    "Database"
    "Containers & Orchestration"
    "Cloud & Infrastructure"
    "Build Tools"
    "Package Managers"
    "Utilities"
  )

  IFS=':' read -ra dirs <<< "$PATH"
  for dir in "${dirs[@]}"; do
    [[ -d "$dir" ]] || continue
    for file in "$dir"/*; do
      [[ -f "$file" && -x "$file" ]] || continue
      local name
      name=$(basename "$file")
      [[ -n "${seen[$name]:-}" ]] && continue
      seen[$name]=1

      local category
      category=$(categorize "$name")
      [[ -z "$category" ]] && continue

      buffers[$category]+="$name"$'\n'
    done
  done

  local first=1
  for cat in "${category_order[@]}"; do
    [[ -z "${buffers[$cat]:-}" ]] && continue
    [[ $first -eq 0 ]] && echo
    first=0
    echo "── $cat"
    printf '%s' "${buffers[$cat]}" | sort | sed 's/^/  /'
  done
}

main
```

Write this to `~/.local/bin/list-clis`.
